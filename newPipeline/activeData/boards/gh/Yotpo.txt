		// Function to retrieve the value of a cookie by name
		function getCookie(cookieName) {
			const name = cookieName + "=";
			const decodedCookie = decodeURIComponent(document.cookie);
			const cookieArray = decodedCookie.split(';');
			for (let i = 0; i < cookieArray.length; i++) {
				let cookie = cookieArray[i];
				while (cookie.charAt(0) === ' ') {
					cookie = cookie.substring(1);
				}
				if (cookie.indexOf(name) === 0) {
					return cookie.substring(name.length, cookie.length);
				}
			}
			return "";
		}

                            <h3>Yotpo’s Recruiting Process</h3>
                            <p>Yotpo’s most important asset? Our team of dedicated, creative, and smart people! While the exact recruiting process varies by location, below you’ll find the most common steps to expect. </p>
                        </div>
                        <div class="interview-steps">
                                                            <div class="step">
                                    <div class="front-step">
                                        <span>01</span>
                                        <p>Application and Resume Screening</p>
                                    </div>
                                    <div class="back-step">Expect a confirmation email once your application is received. Our team reviews every resume typically within a week.</div>
                                </div>
                                                            <div class="step">
                                    <div class="front-step">
                                        <span>02</span>
                                        <p>Initial Phone Screen</p>
                                    </div>
                                    <div class="back-step">Your Talent Acquisition Partner will learn more about you, your goals, discuss compensation, and tell you more about Yotpo.</div>
                                </div>
                                                            <div class="step">
                                    <div class="front-step">
                                        <span>03</span>
                                        <p>Hiring Manager Interview</p>
                                    </div>
                                    <div class="back-step">This call is either an in-person or virtual interview with the team leader. Use this time to ask questions and learn more about the role.</div>
                                </div>
                                                            <div class="step">
                                    <div class="front-step">
                                        <span>04</span>
                                        <p>Practical Exercise and Presentation</p>
                                    </div>
                                    <div class="back-step">Some roles may involve a practical exercise to help us evaluate relevant skills, such as coding, problem-solving, writing, etc.</div>
                                </div>
                                                            <div class="step">
                                    <div class="front-step">
                                        <span>05</span>
                                        <p>Meet the team</p>
                                    </div>
                                    <div class="back-step">Meet additional leaders, business partners, team members, and HR. Each interviewer assesses different aspects of your skills.</div>
                                </div>
                                                            <div class="step">
                                    <div class="front-step">
                                        <span>06</span>
                                        <p>Offer</p>
                                    </div>
                                    <div class="back-step">We appreciate all of your time and we’re excited to extend an offer! We’ll share salary details, benefits, and answer questions. </div>
